P4Runtime Switch Connection Management Module.

This module provides a gRPC-based client for communicating with P4Runtime-enabled switches. It handles device configuration, table entry management, counter operations, and packet I/O through a P4Runtime interface.

 44def ShutdownAllSwitchConnections():
 45    """Shutdown all active P4Runtime switch connections.

 54class SwitchConnection(object):
 55    """Client for communicating with P4Runtime-enabled switches via gRPC.
 56    
 57    This class manages the connection to a single P4Runtime device and provides
 58    methods for configuration, table operations, counter reading, and packet I/O.
 59    Automatically registers itself globally for lifecycle management.
 60    
 61    Attributes:
 62        name (str): Friendly name for the switch connection
 63        address (str): gRPC server address in format 'host:port'
 64        device_id (int): Unique device identifier for the switch
 65        p4info (P4Info): Parsed P4Info protobuf from the device
 66        channel (grpc.Channel): gRPC channel to the switch
 67        client_stub (P4RuntimeStub): P4Runtime service stub
 68        requests_stream (IterableQueue): Queue for outgoing StreamChannel messages
 69        stream_msg_resp: Streaming response iterator from StreamChannel
 70    """
 71
 72    def __init__(self, name=None, address='127.0.0.1:50051', device_id=0,
 73                 proto_dump_file=None):
 74        """Initialize a P4Runtime switch connection.
 75        
 76        Args:
 77            name (str, optional): Descriptive name for this connection. Defaults to None.
 78            address (str, optional): gRPC server endpoint. Defaults to '127.0.0.1:50051'.
 79            device_id (int, optional): Device ID for this switch. Defaults to 0.
 80            proto_dump_file (str, optional): Path to file for logging gRPC messages.
 81                If provided, all requests are logged to this file. Defaults to None.
 82        
 83        Note:
 84            This connection is automatically added to the global connections list
 85            for lifecycle management via ShutdownAllSwitchConnections().
 86        """
 87        self.name = name
 88        self.address = address
 89        self.device_id = device_id
 90        self.p4info = None
 91        self.channel = grpc.insecure_channel(self.address)
 92        if proto_dump_file is not None:
 93            interceptor = GrpcRequestLogger(proto_dump_file)
 94            self.channel = grpc.intercept_channel(self.channel, interceptor)
 95        self.client_stub = p4runtime_pb2_grpc.P4RuntimeStub(self.channel)
 96        self.requests_stream = IterableQueue()
 97        self.stream_msg_resp = self.client_stub.StreamChannel(
 98            iter(self.requests_stream))
 99        self.proto_dump_file = proto_dump_file
100        connections.append(self)

120    def shutdown(self):
121        """Gracefully shutdown the switch connection.
122        
123        Closes the request stream and cancels the response stream to cleanly
124        terminate the gRPC bidirectional streaming connection.
125        """
126        self.requests_stream.close()
127        self.stream_msg_resp.cancel()

129    def MasterArbitrationUpdate(self, dry_run=False, **kwargs):
130        """Send master arbitration request to establish primary controller.
131        
132        Acquires controller mastership for this device by sending an arbitration
133        update with election ID. This must be done before any other operations.
134        
135        Args:
136            dry_run (bool, optional): If True, print request without sending.
137                Defaults to False.
138            **kwargs: Additional arguments (unused, for interface compatibility)
139        
140        Returns:
141            StreamMessageResponse: Response from device arbitration process
142        
143        Note:
144            Uses a fixed election ID (high=0, low=1). Only one master can be active.
145        """
146        request = p4runtime_pb2.StreamMessageRequest()
147        request.arbitration.device_id = self.device_id
148        request.arbitration.election_id.high = 0
149        request.arbitration.election_id.low = 1
150
151        if dry_run:
152            print("P4Runtime MasterArbitrationUpdate: ", request)
153        else:
154            self.requests_stream.put(request)
155            for item in self.stream_msg_resp:
156                return item  # just one

158    def SetForwardingPipelineConfig(self, p4info, dry_run=False, **kwargs):
159        """Configure the P4 forwarding pipeline on the device.
160        
161        Loads the P4Info and device configuration onto the switch. This must be
162        called after master arbitration and before any table operations.
163        
164        Args:
165            p4info (P4Info): Parsed P4Info protobuf containing program metadata
166            dry_run (bool, optional): If True, print request without sending.
167                Defaults to False.
168            **kwargs: Device-specific configuration arguments passed to buildDeviceConfig()
169        
170        Raises:
171            grpc.RpcError: If the configuration fails on the device
172        """
173        device_config = self.buildDeviceConfig(**kwargs)
174        request = p4runtime_pb2.SetForwardingPipelineConfigRequest()
175        request.election_id.low = 1
176        request.device_id = self.device_id
177        config = request.config
178
179        config.p4info.CopyFrom(p4info)
180        config.p4_device_config = device_config.SerializeToString()
181
182        request.action = p4runtime_pb2.SetForwardingPipelineConfigRequest.VERIFY_AND_COMMIT
183        if dry_run:
184            print("P4Runtime SetForwardingPipelineConfig:", request)
185        else:
186            self.client_stub.SetForwardingPipelineConfig(request)

188    def WriteTableEntry(self, table_entry, dry_run=False):
189        """Insert or modify a forwarding table entry.
190        
191        Sends a table entry to the device. Automatically detects whether this is
192        a default action (MODIFY) or a regular entry (INSERT).
193        
194        Args:
195            table_entry (TableEntry): Table entry protobuf to write
196            dry_run (bool, optional): If True, print request without sending.
197                Defaults to False.
198        
199        Raises:
200            grpc.RpcError: If the write operation fails on the device
201        """
202        request = p4runtime_pb2.WriteRequest()
203        request.device_id = self.device_id
204        request.election_id.low = 1
205        update = request.updates.add()
206        if table_entry.is_default_action:
207            update.type = p4runtime_pb2.Update.MODIFY
208        else:
209            update.type = p4runtime_pb2.Update.INSERT
210        update.entity.table_entry.CopyFrom(table_entry)
211        if dry_run:
212            print("P4Runtime Write:", request)
213        else:
214            self.client_stub.Write(request)

235            table_entry.table_id = table_id
236        else:
237            table_entry.table_id = 0
238        if dry_run:
239            print("P4Runtime Read:", request)
240        else:
241            for response in self.client_stub.Read(request):
242                yield response

347class GrpcRequestLogger(grpc.UnaryUnaryClientInterceptor,
348                        grpc.UnaryStreamClientInterceptor):
349    """gRPC interceptor for logging all requests to a file.
350    
351    Captures all unary and streaming P4Runtime requests and logs them to a file
352    with timestamps. Useful for debugging and auditing switch communications.
353    
354    Attributes:
355        log_file (str): Path to the log file for storing request messages
356    """
357
358    def __init__(self, log_file):
359        """Initialize the gRPC request logger.
360        
361        Args:
362            log_file (str): Path to the file where requests will be logged.
363                The file is cleared/created on initialization.
364        """
365        self.log_file = log_file
366        with open(self.log_file, 'w') as f:
367            # Clear content if it exists.
368            f.write("")
369
370    def log_message(self, method_name, body):
371        """Log a single gRPC request message.
372        
373        Appends a timestamped message to the log file. Large messages exceeding
374        MSG_LOG_MAX_LEN are truncated to avoid log file bloat.
375        
376        Args:
377            method_name (str): Name of the gRPC method being called
378            body: Request protobuf message to log
379        """
380        with open(self.log_file, 'a') as f:
381            ts = datetime.utcnow().strftime('%Y-%m-%d %H:%M:%S.%f')[:-3]
382            msg = str(body)
383            f.write("\n[%s] %s\n---\n" % (ts, method_name))
384            if len(msg) < MSG_LOG_MAX_LEN:
385                f.write(str(body))
386            else:
387                f.write("Message too long (%d bytes)! Skipping log...\n" % len(msg))
388            f.write('---\n')
389
390    def intercept_unary_unary(self, continuation, client_call_details, request):
391        """Intercept unary-unary gRPC calls and log them.
392        
393        Args:
394            continuation: Callback to invoke the actual RPC call
395            client_call_details: Details about the RPC call being made
396            request: Request protobuf message
397        
398        Returns:
399            Response from the actual RPC call
400        """
401        self.log_message(client_call_details.method, request)
402        return continuation(client_call_details, request)
403
404    def intercept_unary_stream(self, continuation, client_call_details, request):
405        """Intercept unary-stream gRPC calls and log them.
406        
407        Args:
408            continuation: Callback to invoke the actual RPC call
409            client_call_details: Details about the RPC call being made
410            request: Request protobuf message
411        
412        Returns:
413            Stream response from the actual RPC call
414        """
415        self.log_message(client_call_details.method, request)
416        return continuation(client_call_details, request)

419class IterableQueue(Queue):
420    """Queue wrapper that supports iteration protocol.
421    
422    Allows a queue to be iterated using a for loop or iterator pattern.
423    Iteration terminates when a sentinel value is encountered.
424    
425    Attributes:
426        _sentinel: Special marker object signaling iteration end
427    """
428    _sentinel = object()
429
430    def __iter__(self):
431        """Enable iteration over queue items.
432        
433        Returns:
434            Iterator that yields items from the queue until sentinel is encountered
435        """
436        return iter(self.get, self._sentinel)
437
438    def close(self):
439        """Signal end of iteration by putting sentinel value.
440        
441        Inserts the sentinel object into the queue, which terminates any active
442        iteration when encountered.
443        """
444        self.put(self._sentinel)
